function doubleBalls(){
    while (ballsPresent()){
        takeBall();
        move();
        putBall();
        putBall();
        turnAround();
        move();
        turnAround();
    }
}
function putBack(){
    while (ballsPresent()){
        takeBall();
        move();
        putBall();
        //putBall();
        turnAround();
        move();
        turnAround();
    }
}
function dance(){
    turnAround();
    turnAround();
}
