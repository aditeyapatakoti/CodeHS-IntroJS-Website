1.11.6 Right vs. Left Square

  
function start() {
    if (facingEast()) {
        makeLeftSquare();
    } else {
        makeRightSquare();
    }
}

function makeLeftSquare() {
    for (var i = 0; i < 4; i++) {
        putBall();
        move();
        turnLeft();
    }
}

function makeRightSquare() {
    for (var i = 0; i < 4; i++) {
        putBall();
        move();
        turnRight();
    }
}
